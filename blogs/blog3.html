Binary Search Tree (DSA) With C language

Defination: A binary search tree (BST) is a binary tree where each node has a Comparable key (and an associated value) and satisfies the restriction that the key in any node is larger than the keys in all nodes in that node's left subtree and smaller than the keys in all nodes in that node's right subtree.

Properties:

  1. Each data values in it’s left sub-tree is less then the root value.
  2. Each data values in it’s right sub tree is greater than the root value.
  3. Left and Right sub-trees are again Binary search Tree.

Example:

                
    #include <stdio.h>
    #include <conio.h>
    #include <stdlib.h>
    //include all the header file first the create a note.
    struct node
    {
        int data;
        struct node *left;
        struct node *right;
    };
    struct node *root_node = NULL;
    struct node *create_node(int value)
    {
        if (value != (int)NULL)
        {
            struct node *new_node;
            new_node = (struct node *)malloc(sizeof(struct node));
            new_node->data = value;
            new_node->left = NULL;
            new_node->right = NULL;
            return new_node;
        }
        else
        {
            printf("\nInvalid input");
        }
    }
    void insert(struct node *node, int value)
    {
        if (root_node == NULL)
        {
            root_node = create_node(value);
        }
        else
        {
            struct node *temp_node;
            while (node != NULL)
            {
                temp_node = node;
                if (value == node->data)
                {
                    printf("\nCan't insert, already exist");
                    return;
                }
                else if (value < node->data)
                {
                    node = node->left;
                }
                else
                {
                    node = node->right;
                }
            }
            struct node *new = create_node(value);
            if (value < temp_node->data)
            {
                temp_node->left = new;
            }
            else
            {
                temp_node->right = new;
            }
        }
    }
    void pre_order(struct node *node)
    {
        if (node != NULL)
        {
            printf(" %d ", node->data);
            pre_order(node->left);
            pre_order(node->right);
        }
    }
    void in_order(struct node *node)
    {
        if (node != NULL)
        {
            in_order(node->left);
            printf(" %d ", node->data);
            in_order(node->right);
        }
    }
    void post_order(struct node *node)
    {
        if (node != NULL)
        {
            post_order(node->left);
            post_order(node->right);
            printf(" %d ", node->data);
        }
    }
    void traversal()

        printf("\n\n<<=========Traversal Menu==========>");

        printf("\n\n=====>    ");
        switch (getch())

        case '1':
            pre_order(root_node);

        case '4':
            return;
            break;
    
        default:
            printf("\nInvalid input");
        }
    }
    struct node *in_order_predecessor(struct node *node)
    {
        node = node->left;
        while (node->right != NULL)
        {
            node = node->right;
        }
        return node;
    }
    struct node * delete_node(struct node *node, int value)
    {
        struct node *in_pre; //in_order_predecessor input
        if (node == NULL)
        {
            return NULL;
        }
        if (node->left == NULL && node->right == NULL)
        {
            free(node);
            return NULL;
        }
        //searching value from bst
        if (value < node->data)
        {
            node->left = delete_node(node->left, value);
        }
        else if (value > node->data)
        {
            node->right = delete_node(node->right, value);
        }
        //deletion startegy
        else
        {
            in_pre = in_order_predecessor(node);
            node->data = in_pre->data;
            node->left = delete_node(node->left, in_pre->data);
        }
        return node;
    }
    struct node *search(struct node *node, int value)
    {
        while (node != NULL)
        {
            if (value == node->data)
            {
                return node;
            }
            else if (value < node->data)
            {
                node = node->left;
            }
            else
            {
                node = node->right;
            }
        }
        return NULL;
    }
